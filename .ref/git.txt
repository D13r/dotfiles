g a         git add -A

g b         git branch
g ba        git branch -a (all)
g bclean    git branch -d (delete merged branches)

g c         git commit -m
g ca        git commit --amend --no-edit
g ce        git commit --amend
g cls       git grep -i "class\s\+$1\b"
g co        git checkout
g cp        git cherry-pick

g d         git diff
g dc        git diff --cached

g f         git fetch
g fun       git grep -i "function\s\+$1\b"

g g         git grep
g g3        git grep --context=3 (also 'g6' and 'g9')
g ga        git rev-list | grep ... (search git history)
g gc-all    git gc ...
g gf        git ls-files | grep ... (find files by name)
g gi        git grep --ignore-case
g gi3       git grep --ignore-case --context=3 (also 'gi6' and 'gi9')

g in        git log origin/main..
g io        git log --left-right origin/master..HEAD

g l         git log --name-status
g l1        git log --name-status --pretty=...
g lg        git log --graph
g lg1       git log --graph --pretty=...
g ll        git log
g lp        git log --patch
g lpw       git log --patch --ignore-all-space

g m         git merge
g mi        git checkout $1 && git merge <old> && ...
g mt        git mergetool

g out       git log ..origin/main

g p         git push
g pt        git push --tags
g pu        git push -u origin HEAD

g s         git status
g sub       git submodule
g sync      git submodule sync; submodule update --init

g todo      git grep 'TODO\|XXX\|FIXME'
